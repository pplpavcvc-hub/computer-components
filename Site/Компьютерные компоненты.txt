        // –£–ø—Ä–∞–≤–ª–µ–Ω–∏–µ —Å—Ç—Ä–∞–Ω–∏—Ü–∞–º–∏
        function showPage(pageId) {
            const pages = document.querySelectorAll('.page');
            pages.forEach(page => {
                page.classList.remove('active');
            });
            
            const targetPage = document.getElementById(pageId + '-page');
            if (targetPage) {
                targetPage.classList.add('active');
            }
            
            window.scrollTo({ top: 0, behavior: 'smooth' });
            
            // –ü—Ä–∏ –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏–∏ —Å—Ç—Ä–∞–Ω–∏—Ü —Å–±—Ä–∞—Å—ã–≤–∞–µ–º –∫–∞—Ä—É—Å–µ–ª—å –Ω–∞ –ø–µ—Ä–≤—ã–π —Å–ª–∞–π–¥
            if (pageId === 'cpu' || pageId === 'gpu' || pageId === 'ram' || 
                pageId === 'storage' || pageId === 'motherboard' || pageId === 'psu') {
                setTimeout(() => {
                    goToSlide(pageId, 0);
                }, 100);
            }
        }
        
        function showComponent(componentId) {
            showPage(componentId);
        }
        
        function scrollToComponents() {
            const componentsSection = document.getElementById('components');
            if (componentsSection) {
                componentsSection.scrollIntoView({ behavior: 'smooth' });
            }
        }
        
        // –£–ø—Ä–∞–≤–ª–µ–Ω–∏–µ –∫–∞—Ä—É—Å–µ–ª—å—é - —Ç–µ–ø–µ—Ä—å 3 —Å–ª–∞–π–¥–∞ –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–∞
        const carouselData = {
            cpu: { currentIndex: 0, totalSlides: 3 },
            gpu: { currentIndex: 0, totalSlides: 3 },
            ram: { currentIndex: 0, totalSlides: 3 },
            storage: { currentIndex: 0, totalSlides: 3 },
            motherboard: { currentIndex: 0, totalSlides: 3 },
            psu: { currentIndex: 0, totalSlides: 3 }
        };
        
        function updateCarousel(componentId) {
            const data = carouselData[componentId];
            const carousel = document.getElementById(`${componentId}-carousel`).querySelector('.carousel');
            const dots = document.getElementById(`${componentId}-dots`).querySelectorAll('.carousel-dot');
            
            // –û–±–Ω–æ–≤–ª—è–µ–º –ø–æ–∑–∏—Ü–∏—é –∫–∞—Ä—É—Å–µ–ª–∏
            carousel.style.transform = `translateX(-${data.currentIndex * 100}%)`;
            
            // –û–±–Ω–æ–≤–ª—è–µ–º —Ç–æ—á–∫–∏
            dots.forEach((dot, index) => {
                if (index === data.currentIndex) {
                    dot.classList.add('active');
                } else {
                    dot.classList.remove('active');
                }
            });
        }
        
        function nextSlide(componentId) {
            const data = carouselData[componentId];
            data.currentIndex = (data.currentIndex + 1) % data.totalSlides;
            updateCarousel(componentId);
        }
        
        function prevSlide(componentId) {
            const data = carouselData[componentId];
            data.currentIndex = (data.currentIndex - 1 + data.totalSlides) % data.totalSlides;
            updateCarousel(componentId);
        }
        
        function goToSlide(componentId, index) {
            const data = carouselData[componentId];
            if (index >= 0 && index < data.totalSlides) {
                data.currentIndex = index;
                updateCarousel(componentId);
            }
        }
        
        // –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏–µ —Å–ª–∞–π–¥–æ–≤
        let autoSlideInterval;
        
        function startAutoSlide(componentId) {
            if (autoSlideInterval) {
                clearInterval(autoSlideInterval);
            }
            
            autoSlideInterval = setInterval(() => {
                nextSlide(componentId);
            }, 5000);
        }
        
        function stopAutoSlide() {
            if (autoSlideInterval) {
                clearInterval(autoSlideInterval);
            }
        }
        
        // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è

            // –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏–µ —Å–ª–∞–π–¥–æ–≤ –ø—Ä–∏ –Ω–∞–≤–µ–¥–µ–Ω–∏–∏
            const carouselContainers = document.querySelectorAll('.carousel-container');
            carouselContainers.forEach(container => {
                container.addEventListener('mouseenter', () => {
                    stopAutoSlide();
                });
                
                container.addEventListener('mouseleave', () => {
                    const componentId = container.id.replace('-carousel', '');
                    startAutoSlide(componentId);
                });
            });
            
            // –ó–∞–ø—É—Å–∫–∞–µ–º –∞–≤—Ç–æ–ø—Ä–æ–∫—Ä—É—Ç–∫—É –¥–ª—è –∞–∫—Ç–∏–≤–Ω–æ–π –∫–∞—Ä—É—Å–µ–ª–∏
            startAutoSlide('cpu');

        // –û—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –∞–≤—Ç–æ–ø—Ä–æ–∫—Ä—É—Ç–∫—É –ø—Ä–∏ –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏–∏ —Å—Ç—Ä–∞–Ω–∏—Ü
        document.addEventListener('visibilitychange', () => {

                const activePage = document.querySelector('.page.active');
                if (activePage && activePage.id) {
                    const componentId = activePage.id.replace('-page', '');
                    if (carouselData[componentId]) {
                        startAutoSlide(componentId);
                    }
                }
            }
        });
        
        // –û–±—Ä–∞–±–æ—Ç–∫–∞ –∫–ª–∞–≤–∏—à –¥–ª—è —É–ø—Ä–∞–≤–ª–µ–Ω–∏—è –∫–∞—Ä—É—Å–µ–ª—å—é
        document.addEventListener('keydown', (e) => {
            const activePage = document.querySelector('.page.active');
            if (activePage && activePage.id) {
                const componentId = activePage.id.replace('-page', '');
                if (carouselData[componentId]) {
                    if (e.key === 'ArrowLeft') {
                        prevSlide(componentId);
                        e.preventDefault();
                    } else if (e.key === 'ArrowRight') {
                        nextSlide(componentId);
                        e.preventDefault();
                    }
                }
            }
        });
